    Setup and Structure

    // Functions:
function setup() {
  createCanvas(400, 400); // Creates a canvas of 400x400 pixels
}

function draw() {
  background(220); // Sets the background color to light gray (220)
}

// Data Types:
let boolValue = true; // Boolean
let numValue = 42; // Number
let stringValue = "Hello, p5.js!"; // String

// Objects:
let myObject = {
  key1: "value1",
  key2: 42,
  key3: true
}; // Object

// Arrays:
let myArray = [1, 2, 3, 4, 5]; // Array

// Undefined and Null:
let undefinedValue; // Undefined
let nullValue = null; // Null

    Shapes

    Shape Parameters
    Circle (x, y)
    Point (x, y)
    Line (x1, y1, x2, y2)
    Ellipse (x, y, width, height)
    Square (x, y, side)
    Rectangle (x, y, width, height)
    Triangle (x1, y1, x2, y2, x3, y3) 
      // Drawing shapes:
  circle(200, 200, 50); // Circle
  point(100, 100); // Point
  line(300, 300, 400, 200); // Line
  ellipse(100, 300, 80, 120); // Ellipse
  square(300, 100, 50); // Square
  rect(50, 50, 120, 80); // Rectangle
  triangle(200, 300, 250, 200, 300, 300); // Triangle

    Colors

    // Set fill color

// Set stroke color
stroke(r, g, b, alpha);

// Disable stroke

// Disable fill
noFill();
